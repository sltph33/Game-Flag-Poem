<script>
let attemptsLeft = 3;
let scores = [];
let gameOver = false;

let globalConcept = {
  symbol:0, power:0, nation:0, consequence:0, emotion:0
};

function updateHearts(){
  document.getElementById("hearts").innerHTML =
    "❤️".repeat(attemptsLeft) + "🤍".repeat(3 - attemptsLeft);
}
updateHearts();

function checkAnswer(){
  if(gameOver) return;

  const text = document.getElementById("answer").value.toLowerCase();

  const concepts = {
    symbol:["flag","symbol","represents","meaning"],
    power:["power","control","authority"],
    nation:["nation","country","identity"],
    consequence:["result","impact","cause","effect"],
    emotion:["fear","pride","belief"]
  };

  let local = {symbol:0,power:0,nation:0,consequence:0,emotion:0};

  for(let c in concepts){
    for(let w of concepts[c]){
      if(text.includes(w)){
        local[c]=1;
        globalConcept[c]=1;
      }
    }
  }

  let core = Object.values(local).reduce((a,b)=>a+b,0);
  let score = Math.min(100, core * 20);

  scores.push(score);
  attemptsLeft--;
  updateHearts();

  document.getElementById("score").innerText = "Score: " + score;
  document.getElementById("answer").value = "";

  if(attemptsLeft === 0){
    gameOver = true;

    let total = scores.reduce((a,b)=>a+b,0);

    let feedback = "";
    if(globalConcept.power && !globalConcept.consequence)
      feedback = "You understood power but missed consequence.";
    else if(globalConcept.symbol && !globalConcept.emotion)
      feedback = "You saw the symbol but missed emotional meaning.";
    else if(globalConcept.nation && globalConcept.power)
      feedback = "You understood nation and control.";
    else
      feedback = "You partially understood the poem's meaning.";

    document.getElementById("finalScore").innerText = total + "/300";
    document.getElementById("feedback").innerText = feedback;

    document.getElementById("gamePage").classList.remove("active");
    document.getElementById("endPage").classList.add("active");
  }
}

function goPoem(){
  document.getElementById("gamePage").classList.remove("active");
  document.getElementById("poemPage").classList.add("active");
}

function goGame(){
  document.getElementById("poemPage").classList.remove("active");
  document.getElementById("gamePage").classList.add("active");
}

function restart(){
  location.reload();
}
</script>
